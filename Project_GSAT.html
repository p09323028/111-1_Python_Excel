def descdf(varx, df=df):
    gscore = ['chin', 'engl', 'math', 'science', 'society', 'g_total']
    desc = df.groupby(varx)[gscore].mean().dropna().round(2)
    desc['size'] = df.groupby(varx)['g_total'].count().dropna()
    desc.index.name = ''
    desc.columns = ['國文', '英文', '數學', '自然', '社會', '總級分', '樣本大小']
    return desc

def plotfig(varx, xname, vary='g_total', df=df, order=None):
    catego = df[varx].value_counts()
    df[varx] = df[varx].cat.set_categories(list(catego[catego>0].index), ordered=True)
    fig, ax = plt.subplots()
    sns.violinplot(x=vary, y=varx, data=df, ax=ax, order=order)
    yname = {'g_total': '學測總成績', 'chin': '學測國文成績',
             'engl': '學測英文成績', 'math': '學測數學成績',
             'science': '學測自然成績', 'society': '學測社會成績',}
    ax.set_xlabel(yname[vary])
    ax.set_ylabel(xname)
    plt.show()
